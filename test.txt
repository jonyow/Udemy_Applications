    function togglePanel(jFooterPanel, detailsElement) {
       if (jFooterPanel .hasClass("opened")) {
          jFooterPanel
            .removeClass("opened")
            .addClass("closed");
          $(detailsElement).hide();
        } else {
          jFooterPanel
            .removeClass("closed")
            .addClass("opened");
          $(detailsElement).show();
        }
    }

    //Close the site map panel
    function closePanel(jFooterPanel, detailsElement) {
        if (jFooterPanel .hasClass("opened")) {
          jFooterPanel
            .removeClass("opened")
            .addClass("closed");
          $(detailsElement).hide();
        }
    }
